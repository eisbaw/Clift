% Figure: Worked example — C to CSimpl to L1 to proof
\begin{figure}[t]
\centering
\begin{tikzpicture}[
  cbox/.style={draw, rounded corners=3pt, fill=gray!5,
               text width=3.6cm, font=\ttfamily\tiny, align=left,
               inner sep=3pt},
  arrow/.style={->, >=stealth, thick, color=blue!60!black},
  lbl/.style={font=\scriptsize\sffamily\bfseries, text=blue!60!black},
]
  % Row 1
  \node[cbox] (csrc) at (0, 0) {
    \normalfont\scriptsize\textbf{C source}\\[2pt]
    \texttt{void swap(uint32\_t *a,}\\
    \texttt{\ \ \ \ \ \ \ \ \ uint32\_t *b) \{}\\
    \texttt{\ \ uint32\_t t = *a;}\\
    \texttt{\ \ *a = *b;\ \ *b = t;}\\
    \texttt{\}}
  };
  \node[cbox] (csimpl) at (4.3, 0) {
    \normalfont\scriptsize\textbf{CSimpl (L0)}\\[2pt]
    \texttt{seq (guard valid\_a)}\\
    \texttt{\ (seq (basic (read *a))}\\
    \texttt{\ \ (seq (guard valid\_b)}\\
    \texttt{\ \ \ (seq (basic (*a:=*b))}\\
    \texttt{\ \ \ \ (basic (*b:=t)))))}
  };
  \draw[arrow] (csrc.east) -- node[above, lbl] {\tiny import} (csimpl.west);

  % Row 2
  \node[cbox] (l1) at (0, -2.5) {
    \normalfont\scriptsize\textbf{L1 monadic}\\[2pt]
    \texttt{do guard (valid a)}\\
    \texttt{\ \ let t := hVal heap a}\\
    \texttt{\ \ guard (valid b)}\\
    \texttt{\ \ modify (update a (hVal b))}\\
    \texttt{\ \ modify (update b t)}
  };
  \node[cbox] (prf) at (4.3, -2.5) {
    \normalfont\scriptsize\textbf{User proof}\\[2pt]
    \texttt{validHoare}\\
    \texttt{\ (valid a $\wedge$ valid b}\\
    \texttt{\ \ $\wedge$ disjoint a b)}\\
    \texttt{\ swap\_l1}\\
    \texttt{\ (hVal a = old\_b}\\
    \texttt{\ \ $\wedge$ hVal b = old\_a)}
  };
  \draw[arrow] (csimpl.south) -- node[right, lbl] {\tiny L1} (l1.north -| csimpl.south);
  \draw[arrow] (l1.east) -- node[above, lbl] {\tiny spec} (prf.west);

  % Diagonal arrow from csimpl to l1
  \draw[arrow, dashed, gray] (csimpl.south west) -- (l1.north east);

\end{tikzpicture}
\caption{Worked example: \texttt{swap} through the pipeline. C source
is parsed into CSimpl, lifted to L1 monadic form, then proven correct
with heap disjointness.}
\label{fig:worked}
\end{figure}
